#### 冒泡排序
```javascript
var arr = [1,2,4,3,6,9,7,6,8,5];
function incSort(arr){
    let len = arr.length;
    var mid;
    for(var i = len - 1; i > 0; i--){
        for(var j = 0; j < i; j++){
            if(arr[j] > arr[j+1]){
                mid = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = mid;
            }
        }
    }
    return arr;
}
console.log(incSort(arr));
```
